sargasso$Temp <- as.numeric(sargasso$Temp)

# lm1t <- lm(DO ~ Distance_coast + Temp + Plume + Season + Locality, data = sargasso)


dat <- predict_response(lm1, terms = c( "Distance_coast", "Plume", "Season", "Locality" ))
